\begin{piosenka}[4mm]{Zagubieni żeglarze -- Stara Kuźnia}
	
Czas wypłynąć na morze, już czas, & h A \\
Pożeglować w przestworza do gwiazd,	& e G A \\
Czas nareszcie wyruszyć gdzieś stąd, & h A \\
By po trudach odszukać swój ląd. & G h A \\[\zwrotkaspace]

\refrenspace Płyniemy, żeglujemy, po wzburzonej wodzie, & D A G A \\
\refrenspace Zziębnięci, przemoczeni, w deszczu, wietrze, chłodzie, & D A G A \\
\refrenspace Oczy w dal zapatrzone, słońce drogę pokaże, & D A h fis \\
\refrenspace Jaki ląd odkryjemy -- zagubieni żeglarze. & h fis A h \\[\zwrotkaspace]

Czas zawinąć do portu, lecz gdzie, & h A \\
Wszystkie porty ukryte we mgle,	& e G A \\
Czas się znaleźć nareszcie u bram, & h A \\
W swej przystani i zostać już tam. & G h A \\[\zwrotkaspace]

\refrenspace Płyniemy, żeglujemy, po spienionej fali, & D A G A \\
\refrenspace Szukamy własnej ziemi w bezimiennej dali, & D A G A \\
\refrenspace Już nas jest tak niewielu, czas zaciera nam twarze, & D A h fis \\
\refrenspace Czy dotrzemy do celu - zagubieni żeglarze. & h fis A h \\[\zwrotkaspace]

Wciąż błądzimy po morzach od lat, & h A \\
Nie liczymy ni zysków, ni strat, & e G A \\	
Jeszcze tylko czasami się śni, & h A \\
Raj szukany wśród nocy i dni. & G h A \\[\zwrotkaspace]

\refrenspace Płyniemy, żeglujemy, unoszą nas fale, & D A G A \\
\refrenspace Zgubioną kiedyś ziemię niełatwo odnaleźć, & D A G A \\
\refrenspace Lądu nie odkryjemy, los nas srogo ukarze, & D A h fis \\
\refrenspace Błądzić wiecznie będziemy - zagubieni żeglarze. & h fis A h \\[\zwrotkaspace]	
	
\end{piosenka}
